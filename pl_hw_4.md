```BASIC
10 INPUT "ENTER ROW VALUE: "; ROW
20 INPUT "ENTER COL VALUE: "; COL
30 IF ROW = 0 OR COL = 0 THEN GOTO 80
40 FOR I = 1 TO ROW * 2 + 1
50 IF INT(I / 2) * 2 <> I THEN GOSUB 100
60 IF INT(I / 2) * 2 = I THEN GOSUB 200
70 NEXT I
80 END

100 FOR J = 1 TO COL
110 PRINT "+--";
120 NEXT J
130 PRINT "+"
140 RETURN

200 FOR J = 1 TO COL
210 PRINT "|  ";
220 NEXT J
230 PRINT "|"
240 RETURN
```
